SELECT A.*,
ROWNUM AS RNUM,
COUNT(*) OVER() AS TOTCNT
FROM (
SELECT * FROM page_prac
ORDER BY id DESC) A ) 
WHERE
RNUM >= 40 AND RNUM <= 50;
